12. Write a program to display prime numbers between 50 to 60.


for num in range(50, 61):
    is_prime = True
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            is_prime = False
            break
    if is_prime:
        print(num)

14. Write a program to display Armstrong numbers.


for num in range(100, 1000):
    digits = list(map(int, str(num)))
    if sum(d**3 for d in digits) == num:
        print(num)

19. Write a program to implement ladders and snake game.


# Ladders and Snakes implementation
# Basic logic: positions affected by ladders/snakes
ladders = {3: 22, 5: 8, 11: 26, 20: 29}
snakes = {17: 4, 19: 7, 21: 9, 27: 1}

position = 0
while position < 30:
    dice = int(input("Roll the dice (1-6): "))
    position += dice
    position = ladders.get(position, position)
    position = snakes.get(position, position)
    print(f"Current Position: {position}")
