1. Write a C# program to check whether a number is even or odd


using System;

class Program {
    public static void CheckEvenOdd(int number) {
        if (number % 2 == 0) {
            Console.WriteLine(number + " is even.");
        } else {
            Console.WriteLine(number + " is odd.");
        }
    }

    static void Main(string[] args) {
        CheckEvenOdd(7);
    }
}

2. Write a C# program to calculate the factorial of a number


using System;

class Program {
    public static int Factorial(int n) {
        if (n == 0 || n == 1) {
            return 1;
        }
        return n * Factorial(n - 1);
    }

    static void Main(string[] args) {
        Console.WriteLine(Factorial(5));
    }
}

3. Create a C# program to reverse a number


using System;

class Program {
    public static void ReverseNumber(int num) {
        int reversed = 0;
        while (num != 0) {
            reversed = reversed * 10 + num % 10;
            num /= 10;
        }
        Console.WriteLine("Reversed Number: " + reversed);
    }

    static void Main(string[] args) {
        ReverseNumber(12345);
    }
}

4. Write a C# program to find the largest element in an array


using System;

class Program {
    public static int FindLargest(int[] arr) {
        int largest = arr[0];
        for (int i = 1; i < arr.Length; i++) {
            if (arr[i] > largest) {
                largest = arr[i];
            }
        }
        return largest;
    }

    static void Main(string[] args) {
        int[] numbers = { 10, 20, 5, 40, 15 };
        Console.WriteLine("Largest Element: " + FindLargest(numbers));
    }
}

5. Create a C# program to check if a number is prime


using System;

class Program {
    public static bool IsPrime(int n) {
        if (n <= 1) {
            return false;
        }
        for (int i = 2; i <= Math.Sqrt(n); i++) {
            if (n % i == 0) {
                return false;
            }
        }
        return true;
    }

    static void Main(string[] args) {
        Console.WriteLine(IsPrime(29));
    }
}
