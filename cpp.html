1. Write a C++ program to check whether a number is even or odd


#include 
using namespace std;

void checkEvenOdd(int number) {
    if (number % 2 == 0) {
        cout << number << " is even." << endl;
    } else {
        cout << number << " is odd." << endl;
    }
}

2. Write a C++ program to calculate the factorial of a number


#include 
using namespace std;

int factorial(int n) {
    if (n == 0 || n == 1) {
        return 1;
    }
    return n * factorial(n - 1);
}

3. Create a C++ program to reverse a number


#include 
using namespace std;

void reverseNumber(int num) {
    int reversed = 0;
    while (num != 0) {
        reversed = reversed * 10 + num % 10;
        num /= 10;
    }
    cout << "Reversed Number: " << reversed << endl;
}

4. Write a C++ program to find the largest element in an array


#include 
using namespace std;

int findLargest(int arr[], int size) {
    int largest = arr[0];
    for (int i = 1; i < size; i++) {
        if (arr[i] > largest) {
            largest = arr[i];
        }
    }
    return largest;
}

5. Create a C++ program to check if a number is prime


#include 
using namespace std;

bool isPrime(int n) {
    if (n <= 1) {
        return false; // not prime
    }
    for (int i = 2; i <= n / 2; i++) {
        if (n % i == 0) {
            return false; // not prime
        }
    }
    return true; // prime
}
